Valores

- grid-area-name

- .

- none

                  grid-area: header;
                }

                  grid-area: sidebar;
                }
                .item-d {
                  grid-area: footer;
                }

                .contenedor {
                  grid-template-columns: 1fr 1fr 1fr 1fr;
                  grid-template-rows: auto;
                  grid-template-areas: 
                    "header header header header"
                    "main main . sidebar"
                    "footer footer footer footer";

- column

- dense

                  grid-auto-flow: row | column | row dense | column dense
